Начало программы
    Ввод первого числа
    Ввод второго числа
    Ввод третьего числа
    
    Сумма = Первое число + Второе число + Третье число
    
    Вывод "Сумма трех чисел: ", Сумма
Конец программы


Реализация
const первое_число = parseFloat(prompt("Введите первое число: "));
const второе_число = parseFloat(prompt("Введите второе число: "));
const третье_число = parseFloat(prompt("Введите третье число: "));


const сумма = первое_число + второе_число + третье_число;


console.log("Сумма трех чисел:", сумма);
